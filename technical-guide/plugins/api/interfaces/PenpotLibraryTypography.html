Interface PenpotLibraryTypography

Represents a typography element from a library in Penpot. This interface extends PenpotLibraryElement and includes properties specific to typography elements.

interface PenpotLibraryTypography {
    getPluginData(key): string;
    setPluginData(key, value): void;
    getPluginDataKeys(): string[];

    textTransform?: null | "uppercase" | "capitalize" | "lowercase";

    applyToTextRange(range): void;
    setFont(font, variant?): void;
}
